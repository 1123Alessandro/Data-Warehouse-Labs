TABLE student_enrollment_facts {
    stud_id         INT [pk]
    adm_id          INT [pk]
    coll_id         INT [pk]
    crs_id          INT [pk]

}

TABLE admission {
    adm_id          INT [primary key]
    adm_year        DATE
    adm_sem         INT
    adm_pgm_name    VARCHAR

TABLE college {
